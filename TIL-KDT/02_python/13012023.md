### TMDB
```python
import requests, pprint

BASE_URL = 'https://api.themoviedb.org/3'
path = '/movie/popular'
params = {
		  'api_key': '',
		  'language': 'ko-KR',
		  'region': 'KR',
}

response = requests.get(BASE_ULR+path, params=params).json()
pprint(response)
pprint(response.get('results')[0])

```
